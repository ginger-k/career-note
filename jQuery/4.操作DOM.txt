思想: 操作之前,必先选中

1.获取文档对象
    children([expr]) -- 获取指定的子元素
    find(expr) -- 获取指定的后代元素
    parents([expr]) -- 获得祖辈元素
    parent() -- 获取父元素
    next([expr]) -- 获取下一个兄弟元素
    prev([expr]) -- 获取前一个兄弟元素
    siblings([expr]) -- 获取所有兄弟元素

2.操作文档对象
    创建元素
        $(elements) //dom对象转换为jquery对象
	$(html) //标签文本转换为jquery
    遍历
	$nodes.each(callback)
    内部插入
	$node.append($newNode) -- 结尾处
	$newNode.appendTo($node)
	$node.prepend($newNode) -- 开始处
	$newNode.prependTo($node)
    外部插入
	$node.after($newNode) -- 元素之后插入
	$newNode.insertAfter($node)
	$node.before($newNode) -- 元素之前插入
	$newNode.insertBefore($node)
    删除节点
	$node.empty() -- 删除node的所有子元素,包括文本
	$node.remove([expr]) -- 删除满足某个表达式的节点node

3.操作文档对象的属性
    attr(name) -- 获取名为name的属性的值
    attr(key,value) -- 设置属性名为key,值为value
    attr(properties) -- 设置属性 {key1:value1,key2:value2 ... }
    removeAttr(name) -- 从元素中删除属性名为name的属性
    prop(name|properties|key,value|fn)
注意:可以设置不存在的属性

4.操作文档对象的css
    attr("style",value) -- 通过attr属性设置获取style属性
    css(name) -- 获取名为name的css样式属性的值
    css(name,value) -- 设置css样式属性名为key,值为value
    css(properties) -- 设置cssuangs属性 {key1:value1,key2:value2 ... }
    addClass(class) -- 添加一个class属性
    removeClass([class]) -- 移除一个class属性
    toggleClass(class) -- 如果存在(不存在)就删除(添加)一个class属性

5.设置或获取HTML,文本和值
    html() -- 获取和设置某个元素中html内容
    html(val)
    text() -- 获取和设置某个元素中的文本内容
    text(val)
    val() -- 获取和设置文本框,下拉列表框,单选框选中的元素值
    val([val|fn|arr]) 
注意: 获取radio,checkbox,select的值的时候要用each(callback)遍历
	$("input[name='gender']:checked").each(function(){alert($(this).val());});
	$("input[name='hobby']:checked").each(function(){alert($(this).val());});
	$("select[name=city] option:selected").each(function(){alert($(this).val());});
      设置radio,checkbox,select的时候,val(arr)的参数是String类型的arr
	$("input[name='gender']").val(["male"]);
	$("input[name='hobby']").val(["coding","swimming","dancing"]);
	$("#city").val(["sh"]);

6.对象访问
    each(callback) 
    size() 
    length  
    get([index]) 
    index([selector|element])
