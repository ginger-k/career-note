页面载入
	ready(fn) -- 网页中所有dom结构绘制完毕后就执行,
		     可能dom元素关联的东西并没有加载完
		     能同时编写多个
		$(document).ready(function(){
		    // 在这里写你的代码...
		});
		    可以简写为:
		$(function(){
		    // 在这里写你的代码...
		});

事件处理
	bind(type,[data],fn) -- 给元素绑定事件,元素必须已经存在
	unbind(type,[data|fn]]) 
	one(type,[data],fn) -- 给元素绑定一个一次性事件
	trigger(type,[data]) -- 在匹配的元素上触发某类事件

事件委派
	live(type,[data],fn) -- 给元素绑定事件,元素可以是以后添加进来
	die(type,[fn])
	on(events,[selector],[data],fn) -- 给元素绑定事件,元素可以是以后添加进来
		$(document).on('blur', 'input[name="password"]', function(e){
				console.log($(e.currentTarget).val());
			}
		);

事件切换
	hover([over,]out) -- 模拟光标悬停事件,
			当光标移动到元素上时，会触发指定的第一个函数,
			当光标移出这个元素时，会触发指定的第二个函数. 
	toggle([speed],[easing],[fn]) -- 模拟鼠标连续单击事件
			第一次单击元素，触发指定的第一个函数，
			当再一次单击同一个元素时，则触发指定的第二个函数，
			如果有更多个函数，则依次触发，直到最后一个
